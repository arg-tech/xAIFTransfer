// Scrapes all corpora shortnames and outputs them in a single array
// Execute in browser in the console window on http://corpora.aifdb.org/

// Gets an array of objects containing links to each corpora
const corporaItems = document.querySelectorAll('.nodeset.clearfix > .corporalink');

// Extracts links to individual corpora from each array object
const corporaLinks = [...corporaItems].map(corpora => corpora.href)

// Extracts shortnames from each link in array by stripping away the prefix
const shortNames = [...corporaLinks].map(function(link) {
    return link.replace(new RegExp("^" + 'http://corpora.aifdb.org/'), '');
});

// Prints out the resulting array
console.log(JSON.stringify(shortNames));
